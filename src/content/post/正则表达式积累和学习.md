# 正则表达式积累和学习

```javascript
function formatNumberWithCommas(number) {
  // 将数字转换为字符串
  let numStr = number.toString()

  // 使用正则表达式在逗号前插入千分位分隔符
  numStr = numStr.replace(/\B(?=(\d{3})+(?!\d))/g, ",")

  return numStr
}

// 示例
let formattedNumber = formatNumberWithCommas(1234567)
console.log(formattedNumber) // 输出 "1,234,567"
```
